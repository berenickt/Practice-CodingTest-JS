# 문자열을 배열로

```js
문자열.split();
```

# 배열

```js
const 배열 = ['a', 'b', 'c'];

// **** 배열 맨뒤 값을 추가, push
배열.push('d');
배열[0] = 'e';

// **** 값만 알고, 인덱스를 모른다면, indexOf
const 인덱스 = 배열.indexOf('b');

// **** 원하는 위치에 값을 추가, 삭제할 떄는 splice
// splice(인덱스, 0, 값)으로 원하는 위치에 추가
// splice(인덱스, 1)로 배열에 값을 삭제
배열.splice(인덱스, 1);
```
